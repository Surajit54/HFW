@app.route("/upload_notice", methods=["GET", "POST"]) def upload_notice(): if request.method == "POST": memo = request.form.get("memo") file = request.files["notice"] if file: filename = file.filename filepath = os.path.join(app.config["UPLOAD_FOLDER"], filename) file.save(filepath) upload_date = datetime.now().strftime("%d-%m-%Y") conn = sqlite3.connect(DB) c = conn.cursor() c.execute( "INSERT INTO notices (memo, filename, upload_date) VALUES (?,?,?)", (memo, filename, upload_date) ) conn.commit() conn.close() # ✅ SUCCESS POPUP MESSAGE flash("Notice Uploaded Successfully ✅") return redirect(url_for("upload_notice")) return render_template("upload_notice.html")
